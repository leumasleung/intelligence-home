/** * 取得巴士到站時間 (主入口函式，一次打包所有路線) */ function getAllBusData() { return { "680": getJointBus680Eta(), "682B": getCitybusEta682B() }; } // ---------------------------------------------------- // 1. 680 聯營線 (合併 KMB + CTB、自動去重並標註中文公司名) // 九巴站牌: A295FD27A4E5A7C5 [小瀝源草地滾球場 (ST530)] // 城巴站牌: 003080 [小瀝源草地滾球場, 小瀝源路] (dir: I) // ---------------------------------------------------- function getJointBus680Eta() { var rawEtaItems = []; var now = new Date(); // 1. 抓取 九巴 KMB (使用正確的 16 位元 Hash Stop ID) try { var kmbUrl = "https://data.etabus.gov.hk/v1/transport/kmb/stop-eta/A295FD27A4E5A7C5"; var kmbRes = UrlFetchApp.fetch(kmbUrl, { muteHttpExceptions: true }); if (kmbRes.getResponseCode() === 200) { var kmbJson = JSON.parse(kmbRes.getContentText()); if (kmbJson.data && Array.isArray(kmbJson.data)) { kmbJson.data.forEach(function(item) { // 只篩選 680 路線且有時間資料的班次 if (item.route === "680" && item.eta) { var t = new Date(item.eta); if (t >= now) { rawEtaItems.push({ time: t, co: "九巴" }); } } }); } } } catch (e) { console.log("KMB 680 Error: " + e.toString()); } // 2. 抓取 城巴 CTB (Stop: 003080, Route: 680, Dir: I) try { var ctbUrl = "https://rt.data.gov.hk/v2/transport/citybus/eta/CTB/003080/680"; var ctbRes = UrlFetchApp.fetch(ctbUrl, { muteHttpExceptions: true }); if (ctbRes.getResponseCode() === 200) { var ctbJson = JSON.parse(ctbRes.getContentText()); if (ctbJson.data && Array.isArray(ctbJson.data)) { ctbJson.data.forEach(function(item) { if (item.dir === "I" && item.eta) { var t = new Date(item.eta); if (t >= now) { rawEtaItems.push({ time: t, co: "城巴" }); } } }); } } } catch (e) { console.log("CTB 680 Error: " + e.toString()); } // 3. 依時間先後順序排序 (由近到遠) rawEtaItems.sort(function(a, b) { return a.time - b.time; }); // 4. 計算剩餘分鐘數 + 【自動去重機制】 var etaList = []; var lastAddedTimeMs = null; // 紀錄上一班的時間 (毫秒) for (var i = 0; i < rawEtaItems.length; i++) { var item = rawEtaItems[i]; var currentTimeMs = item.time.getTime(); var diffMin = Math.round((item.time - now) / 60000); // 去重邏輯：如果與上一班抵達時間相差小於 60 秒 (60000 ms)，視為同一班車並自動跳過 if (lastAddedTimeMs !== null && Math.abs(currentTimeMs - lastAddedTimeMs) < 60000) { continue; } if (diffMin >= 0) { etaList.push({ "min": diffMin, "co": item.co }); lastAddedTimeMs = currentTimeMs; // 更新最後加入班次的時間 } if (etaList.length >= 3) break; // 最多取前 3 班 } return etaList; } // ---------------------------------------------------- // 2. 城巴 682B (Stop: 002051, dir: O) // ---------------------------------------------------- function getCitybusEta682B() { var etaList = []; try { var url = "https://rt.data.gov.hk/v2/transport/citybus/eta/CTB/002051/682B"; var response = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); if (response.getResponseCode() === 200) { var json = JSON.parse(response.getContentText()); if (json.data && Array.isArray(json.data)) { var now = new Date(); json.data.forEach(function(item) { if (item.dir === "O" && item.eta) { var etaTime = new Date(item.eta); var diffMin = Math.round((etaTime - now) / 60000); if (diffMin >= 0) { etaList.push({ "min": diffMin, "co": "城巴" }); } } }); } } } catch (e) { console.log("Citybus 682B Error: " + e.toString()); } // 按抵達分鐘數排序，並取前 3 班 etaList.sort(function(a, b) { return a.min - b.min; }); return etaList.slice(0, 3); }

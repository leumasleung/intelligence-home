// 補助函式：香港天文台天氣 (修正匹配邏輯) function getHKOWeather() { var weatherResult = { "shatin": { "temp": "--", "rh": "--", "icon": "☀️" }, "quarry_bay": { "temp": "--", "rh": "--", "icon": "☀️" }, "tin_hau": { "temp": "--", "rh": "--", "icon": "☀️" } }; try { var url = "https://data.weather.gov.hk/weatherAPI/opendata/weather.php?dataType=rhrread&lang=tc"; var response = UrlFetchApp.fetch(url, {muteHttpExceptions: true}); var json = JSON.parse(response.getContentText()); if (json.temperature && json.temperature.data) { json.temperature.data.forEach(function(item) { // 1. 沙田：直接抓「沙田」 if (item.place === "沙田") { weatherResult.shatin.temp = item.value; } // 2. 鰂魚涌 (東區)：天文台東區站點包含「北角」、「筲箕灣」或「太古城」 if (item.place === "北角" || item.place === "筲箕灣" || item.place === "東區") { weatherResult.quarry_bay.temp = item.value; } // 3. 天后 (中西區/灣仔)：天文台近天后站點包含「跑馬地」、「灣仔」、「香港公園」或「中西區」 if (item.place === "跑馬地" || item.place === "香港公園" || item.place === "中西區" || item.place === "灣仔") { weatherResult.tin_hau.temp = item.value; } }); } // 濕度取全港平均 if (json.humidity && json.humidity.data && json.humidity.data.length > 0) { var mainRh = json.humidity.data[0].value; weatherResult.shatin.rh = mainRh; weatherResult.quarry_bay.rh = mainRh; weatherResult.tin_hau.rh = mainRh; } // 天氣圖示 var iconCode = (json.icon && json.icon.length > 0) ? json.icon[0] : 50; var simpleIcon = mapHKOIcon(iconCode); weatherResult.shatin.icon = simpleIcon; weatherResult.quarry_bay.icon = simpleIcon; weatherResult.tin_hau.icon = simpleIcon; } catch (e) { console.log("Weather API Error: " + e); } return weatherResult; }
